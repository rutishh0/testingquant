API Reference

POST /v1/construction/preprocess
Preprocess Transaction

Creates a request to fetch metadata needed for transaction construction.

Request Example:
{
  "network_identifier": {
    "blockchain": "ethereum",
    "network": "mainnet"
  },
  "operations": [
    {
      "operation_identifier": {
        "index": 0
      },
      "type": "TRANSFER",
      "account": {
        "address": "0x1234567890123456789012345678901234567890"
      },
      "amount": {
        "value": "1000000000000000000",
        "currency": {
          "symbol": "ETH",
          "decimals": 18
        }
      }
    }
  ]
}
Response Example:
{
  "options": {
    "gas_price": "20000000000",
    "gas_limit": "21000"
  }
}
POST /v1/construction/payloads
Create Payloads

Generates an unsigned transaction and payloads to sign.

Request Example:
{
  "network_identifier": {
    "blockchain": "ethereum",
    "network": "mainnet"
  },
  "operations": [...],
  "metadata": {
    "gas_price": "20000000000",
    "gas_limit": "21000"
  }
}
POST /v1/construction/combine
Combine Signatures

Combines an unsigned transaction with signatures.

POST /v1/construction/submit
Submit Transaction

Submits a signed transaction to the blockchain.

POST /v1/account/balance
Get Account Balance

Retrieves the balance of an account for a specific blockchain and network.

Request Example:
{
  "network_identifier": {
    "blockchain": "ethereum",
    "network": "mainnet"
  },
  "account_identifier": {
    "address": "0x1234567890123456789012345678901234567890"
  }
}
Response Example:
{
  "block_identifier": {
    "index": 12345678,
    "hash": "0xabcdef..."
  },
  "balances": [
    {
      "value": "1000000000000000000",
      "currency": {
        "symbol": "ETH",
        "decimals": 18
      }
    }
  ]
}
POST /v1/block
Get Block

Retrieves a block by index or hash.

Request Example:
{
  "network_identifier": {
    "blockchain": "ethereum",
    "network": "mainnet"
  },
  "block_identifier": {
    "index": 12345678
  }
}
POST /v1/transaction
Get Transaction

Retrieves a transaction by its identifier.

Request Example:
{
  "network_identifier": {
    "blockchain": "ethereum",
    "network": "mainnet"
  },
  "transaction_identifier": {
    "hash": "0x1234567890abcdef..."
  }
}

JavaScript
Query Account Balance
const axios = require('axios');

async function getAccountBalance() {
  try {
    const response = await axios.post(
      'http://localhost:8080/v1/account/balance',
      {
        network_identifier: {
          blockchain: 'ethereum',
          network: 'mainnet'
        },
        account_identifier: {
          address: '0x1234567890123456789012345678901234567890'
        }
      },
      {
        headers: {
          'Content-Type': 'application/json',
          'X-API-Key': 'your-api-key-here'
        }
      }
    );
    
    console.log('Account Balance:', response.data);
    return response.data;
  } catch (error) {
    console.error('Error fetching balance:', error.response?.data || error.message);
  }
}

Go
Submit Transaction
package main

import (
	"bytes"
	"encoding/json"
	"fmt"
	"net/http"
)

func main() {
	// Prepare request body
	reqBody, _ := json.Marshal(map[string]interface{}{
		"network_identifier": map[string]string{
			"blockchain": "ethereum",
			"network":    "mainnet",
		},
		"signed_transaction": "0x...", // Your signed transaction
	})

	// Create request
	req, _ := http.NewRequest(
		"POST",
		"http://localhost:8080/v1/construction/submit",
		bytes.NewBuffer(reqBody),
	)

	// Set headers
	req.Header.Set("Content-Type", "application/json")
	req.Header.Set("X-API-Key", "your-api-key-here")

	// Send request
	client := &http.Client{}
	resp, err := client.Do(req)
	if err != nil {
		fmt.Println("Error sending request:", err)
		return
	}
	defer resp.Body.Close()

	// Parse response
	var result map[string]interface{}
	json.NewDecoder(resp.Body).Decode(&result)

	fmt.Println("Transaction submitted:", result)
}
Python
Get Block Information
import requests
import json

def get_block_info(block_index):
    url = "http://localhost:8080/v1/block"
    
    payload = {
        "network_identifier": {
            "blockchain": "ethereum",
            "network": "mainnet"
        },
        "block_identifier": {
            "index": block_index
        }
    }
    
    headers = {
        "Content-Type": "application/json",
        "X-API-Key": "your-api-key-here"
    }
    
    response = requests.post(url, json=payload, headers=headers)
    
    if response.status_code == 200:
        return response.json()
    else:
        print(f"Error: {response.status_code}")
        print(response.text)
        return None

# Get information about block #12345678
block_data = get_block_info(12345678)
print(json.dumps(block_data, indent=2))
Java
Preprocess Transaction
import java.net.http.HttpClient;
import java.net.http.HttpRequest;
import java.net.http.HttpResponse;
import java.net.URI;

public class MeshConnectorExample {
    public static void main(String[] args) {
        try {
            String requestBody = "{"
                + "\"network_identifier\": {"
                + "    \"blockchain\": \"ethereum\","
                + "    \"network\": \"mainnet\""
                + "},"
                + "\"operations\": ["
                + "    {"
                + "        \"operation_identifier\": {"
                + "            \"index\": 0"
                + "        },"
                + "        \"type\": \"TRANSFER\","
                + "        \"account\": {"
                + "            \"address\": \"0x1234567890123456789012345678901234567890\""
                + "        },"
                + "        \"amount\": {"
                + "            \"value\": \"1000000000000000000\","
                + "            \"currency\": {"
                + "                \"symbol\": \"ETH\","
                + "                \"decimals\": 18"
                + "            }"
                + "        }"
                + "    }"
                + "]"
                + "}";

            HttpClient client = HttpClient.newHttpClient();
            HttpRequest request = HttpRequest.newBuilder()
                .uri(URI.create("http://localhost:8080/v1/construction/preprocess"))
                .header("Content-Type", "application/json")
                .header("X-API-Key", "your-api-key-here")
                .POST(HttpRequest.BodyPublishers.ofString(requestBody))
                .build();

            HttpResponse<String> response = client.send(request, HttpResponse.BodyHandlers.ofString());
            System.out.println("Status code: " + response.statusCode());
            System.out.println("Response: " + response.body());

        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}

Get Started

1. Install the Connector

Clone the repository and build the connector:

git clone https://github.com/yourusername/quant-mesh-connector.git
cd quant-mesh-connector
go mod download
go build -o bin/connector cmd/main.go

2. Configure Environment

Create a .env file with your configuration:

cp .env.example .env
# Edit .env with your settings

3. Start the Server

Run the connector:

./bin/connector

Or use Docker:

docker-compose up -d

4. Make Your First API Call

Test the API with a simple health check:

curl http://localhost:8080/health

Expected response:

{
  "status": "healthy",
  "timestamp": 1634567890,
  "version": "1.0.0"
}
